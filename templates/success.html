Ваш ключ: {{ key }}

Используйте этот ключ чтобы отправлять API запросы. Пример:

curl -X POST http://yourapi.com/predict -d "key={{ key }}" -d "data=your_data"
----------------
import requests
import json
url = "http://localhost:5000/predict"
payload = json.dumps({
  "key": "{{ key }}",
  "data": [
    {
      ...
    },
    {
      ...
    }
  ]
})
headers = {'Content-Type': 'application/json'}
response = requests.request("POST", url, headers=headers, data=payload)
print(response.text)
